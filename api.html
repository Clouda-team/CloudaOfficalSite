从Hello World开始


/**
 * Hello World!! 
 */
require("clouda-core");
require("clouda-httpServer");

clouda.setup({
    "resource.appName":"Clouda+",
    "config.clouda-httpServer":{
        autoStart:true,
        defaultAction:function(){
            content = "Hello, Welcome to " + clouda.resource.appName;
            this.sendContent(content);
        }
    }
});
